## React 标准组件

```tsx
import { useState, useEffect } from 'react'
import { Table, Button, Space } from 'antd'
import { getUsers, type User } from '@/api/modules/user'

interface UserListProps {
  onEdit?: (user: User) => void
  onDelete?: (id: string) => void
}

const UserList: React.FC<UserListProps> = ({
  onEdit,
  onDelete
}) => {
  // 1. State 定义
  const [list, setList] = useState<User[]>([])
  const [loading, setLoading] = useState(false)

  // 2. 业务逻辑
  const loadData = async (): Promise<void> => {
    setLoading(true)
    const res = await getUsers({ page: 1, pageSize: 10 })
    setList(res.data)
    setLoading(false)
  }

  // 3. 生命周期
  useEffect(() => {
    loadData()
  }, [])

  // 4. Render
  return (
    <div>
      <Space style={{ marginBottom: 16 }}>
        <Button type="primary">新增用户</Button>
        <Button onClick={loadData}>刷新</Button>
      </Space>

      <Table
        rowKey="id"
        dataSource={list}
        loading={loading}
        columns={[
          { title: '用户名', dataIndex: 'name' },
          { title: '邮箱', dataIndex: 'email' },
          {
            title: '操作',
            render: (_, record) => (
              <Space>
                <Button onClick={() => onEdit?.(record)}>编辑</Button>
                <Button danger onClick={() => onDelete?.(record.id)}>删除</Button>
              </Space>
            )
          }
        ]}
      />
    </div>
  )
}

export default UserList
```

## Vue 3 标准组件

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { ElTable, ElButton, ElMessage } from 'element-plus'
import { getUsers, type User } from '@/api/modules/user'

interface Props {
  onEdit?: (user: User) => void
  onDelete?: (id: string) => void
}

const props = defineProps<Props>()

// 1. State 定义
const list = ref<User[]>([])
const loading = ref(false)

// 2. 业务逻辑
const loadData = async (): Promise<void> => {
  loading.value = true
  const res = await getUsers({ page: 1, pageSize: 10 })
  list.value = res.data
  loading.value = false
}

// 3. 生命周期
onMounted(() => {
  loadData()
})
</script>

<template>
  <div>
    <el-space style="margin-bottom: 16px">
      <el-button type="primary">新增用户</el-button>
      <el-button @click="loadData">刷新</el-button>
    </el-space>

    <el-table
      row-key="id"
      :data="list"
      :loading="loading"
    >
      <el-table-column prop="name" label="用户名" />
      <el-table-column prop="email" label="邮箱" />
      <el-table-column label="操作">
        <template #default="{ row }">
          <el-space>
            <el-button @click="props.onEdit?.(row)">编辑</el-button>
            <el-button type="danger" @click="props.onDelete?.(row.id)">删除</el-button>
          </el-space>
        </template>
      </el-table-column>
    </el-table>
  </div>
</template>
```

## API 请求封装

```typescript
import axios from 'axios'
import { ElMessage } from 'element-plus'

const http = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 30000
})

// ✅ 全局统一错误拦截
http.interceptors.response.use(
  res => res.data,
  error => {
    const message = error.response?.data?.message || '请求失败'
    ElMessage.error(message)
    return Promise.reject(error)
  }
)

export default http
```

> 💡 优先全局拦截，不需要每个请求都 try-catch。
